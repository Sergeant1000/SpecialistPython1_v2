## "Шоколадка"

m = int(input("Ввести длину первой стороны плитки: "))
n = int(input("Ввести длину второй стороны плитки: "))
k = int(input("Ввести количество долек при разломе: "))

if n*m != k:
    if k == m or k == n or n*m-k > n or n*m-k > m:
        print("Да!")
else:
    print("Нет!")
